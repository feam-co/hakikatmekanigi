import { FeamClient } from '@feam/audit-sdk';

const client = new FeamClient({ apiKey: process.env.FEAM_API_KEY });

async function processPrompt(userPrompt) {
    const result = await client.audit.scan({
        text: userPrompt,
        redactMode: 'mask'
    });

    if (result.isSafe) {
        return await openAi.complete(result.cleanText);
    } else {
        throw new Error("PHI ihlali engellendi.");
    }
}
